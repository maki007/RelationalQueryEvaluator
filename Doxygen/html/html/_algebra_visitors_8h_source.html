201  {
202  outputColumns.clear();
203  for (auto it = node->outputJoinColumns.begin(); it != node->outputJoinColumns.end(); ++it)
204  {
205  if (outputColumns.find(it->newColumn) == outputColumns.end())
206  {
207  if (it->input == 0)
208  {
209  if (outputColumns0.find(it->column.name) == outputColumns0.end())
210  {
211  ReportError("Column " + it->column.name + " not found in first input", "join", node->lineNumber);
212  }
213  else
214  {
215  outputColumns[it->newColumn] = ColumnInfo(it->newColumn, outputColumns0[it->column.name].type);
216  outputColumns[it->newColumn].column.id = outputColumns0[it->column.name].column.id;
217  (it)->column.id = outputColumns[it->newColumn].column.id;
218  (it)->type = outputColumns[it->newColumn].type;
219  }
220  }
221  else if (it->input == 1)
222  {
223  if (outputColumns1.find(it->column.name) == outputColumns1.end())
224  {
225  ReportError("Column " + it->column.name + " not found in second input", "join", node->lineNumber);
226  }
227  else
228  {
229  outputColumns[it->newColumn] = ColumnInfo(it->newColumn, outputColumns1[it->column.name].type);
230  outputColumns[it->newColumn].column.id = outputColumns1[it->column.name].column.id;
231  (it)->column.id = outputColumns[it->newColumn].column.id;
232  (it)->type = outputColumns[it->newColumn].type;
233  }
234  }
235  else
236  {
237  throw std::exception("Error");
238  }
239  }
240  else
241  {
242  ReportError("Output column " + it->newColumn + " is allready used", "join", node->lineNumber);
