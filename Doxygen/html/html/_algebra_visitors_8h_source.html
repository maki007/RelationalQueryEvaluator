202  void checkJoinOutPutParameters(std::map<std::string, ColumnInfo> & outputColumns0, std::map<std::string, ColumnInfo> & outputColumns1, T * node)
203  {
204  outputColumns.clear();
205  for (auto it = node->outputJoinColumns.begin(); it != node->outputJoinColumns.end(); ++it)
206  {
207  if (outputColumns.find(it->newColumn) == outputColumns.end())
208  {
209  if (it->input == 0)
210  {
211  if (outputColumns0.find(it->column.name) == outputColumns0.end())
212  {
213  ReportError("Column " + it->column.name + " not found in first input", "join", node->lineNumber);
214  }
215  else
216  {
217  outputColumns[it->newColumn] = ColumnInfo(it->newColumn, outputColumns0[it->column.name].type);
218  outputColumns[it->newColumn].column.id = outputColumns0[it->column.name].column.id;
219  (it)->column.id = outputColumns[it->newColumn].column.id;
220  (it)->type = outputColumns[it->newColumn].type;
221  }
222  }
223  else if (it->input == 1)
224  {
225  if (outputColumns1.find(it->column.name) == outputColumns1.end())
226  {
227  ReportError("Column " + it->column.name + " not found in second input", "join", node->lineNumber);
228  }
229  else
230  {
231  outputColumns[it->newColumn] = ColumnInfo(it->newColumn, outputColumns1[it->column.name].type);
232  outputColumns[it->newColumn].column.id = outputColumns1[it->column.name].column.id;
233  (it)->column.id = outputColumns[it->newColumn].column.id;
234  (it)->type = outputColumns[it->newColumn].type;
235  }
236  }
237  else
238  {
239  throw std::exception("Error");
240  }
241  }
242  else
243  {
244  ReportError("Output column " + it->newColumn + " is allready used", "join", node->lineNumber);
245  }
246  }
